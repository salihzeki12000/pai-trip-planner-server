public final class NaverCarRoute {
    public final Version version;
    public final Summary summary;
    public final Route route[];
    public final String path;
    public final String bound;

    @JsonCreator
    public NaverCarRoute(@JsonProperty("version") Version version, @JsonProperty("summary") Summary summary, @JsonProperty("route") Route[] route, @JsonProperty("path") String path, @JsonProperty("bound") String bound){
        this.version = version;
        this.summary = summary;
        this.route = route;
        this.path = path;
        this.bound = bound;
    }

    public static final class Version {
        public final String route;
        public final String result;

        @JsonCreator
        public Version(@JsonProperty("route") String route, @JsonProperty("result") String result){
            this.route = route;
            this.result = result;
        }
    }

    public static final class Summary {
        public final long returnCode;
        public final long speed;
        public final long searchOption;
        public final String trafficInfo;
        public final long totalDistance;
        public final long totalTime;
        public final String mileage;
        public final long car;
        public final long taxi;
        public final long gasPayPerLiter;
        public final String tollgate;
        public final StartPoint startPoint;
        public final EndPoint endPoint;
        public final RoadSummary roadSummary[];

        @JsonCreator
        public Summary(@JsonProperty("returnCode") long returnCode, @JsonProperty("speed") long speed, @JsonProperty("searchOption") long searchOption, @JsonProperty("trafficInfo") String trafficInfo, @JsonProperty("totalDistance") long totalDistance, @JsonProperty("totalTime") long totalTime, @JsonProperty("mileage") String mileage, @JsonProperty("car") long car, @JsonProperty("taxi") long taxi, @JsonProperty("gasPayPerLiter") long gasPayPerLiter, @JsonProperty("tollgate") String tollgate, @JsonProperty("startPoint") StartPoint startPoint, @JsonProperty("endPoint") EndPoint endPoint, @JsonProperty("roadSummary") RoadSummary[] roadSummary){
            this.returnCode = returnCode;
            this.speed = speed;
            this.searchOption = searchOption;
            this.trafficInfo = trafficInfo;
            this.totalDistance = totalDistance;
            this.totalTime = totalTime;
            this.mileage = mileage;
            this.car = car;
            this.taxi = taxi;
            this.gasPayPerLiter = gasPayPerLiter;
            this.tollgate = tollgate;
            this.startPoint = startPoint;
            this.endPoint = endPoint;
            this.roadSummary = roadSummary;
        }

        public static final class StartPoint {
            public final String name;
            public final long x;
            public final long y;
            public final long px;
            public final long py;
            public final long dist;

            @JsonCreator
            public StartPoint(@JsonProperty("name") String name, @JsonProperty("x") long x, @JsonProperty("y") long y, @JsonProperty("px") long px, @JsonProperty("py") long py, @JsonProperty("dist") long dist){
                this.name = name;
                this.x = x;
                this.y = y;
                this.px = px;
                this.py = py;
                this.dist = dist;
            }
        }

        public static final class EndPoint {
            public final String name;
            public final long x;
            public final long y;
            public final long px;
            public final long py;
            public final long dist;

            @JsonCreator
            public EndPoint(@JsonProperty("name") String name, @JsonProperty("x") long x, @JsonProperty("y") long y, @JsonProperty("px") long px, @JsonProperty("py") long py, @JsonProperty("dist") long dist){
                this.name = name;
                this.x = x;
                this.y = y;
                this.px = px;
                this.py = py;
                this.dist = dist;
            }
        }

        public static final class RoadSummary {
            public final Point point;
            public final String name;
            public final long length;
            public final long congestion;
            public final long speed;

            @JsonCreator
            public RoadSummary(@JsonProperty("point") Point point, @JsonProperty("name") String name, @JsonProperty("length") long length, @JsonProperty("congestion") long congestion, @JsonProperty("speed") long speed){
                this.point = point;
                this.name = name;
                this.length = length;
                this.congestion = congestion;
                this.speed = speed;
            }

            public static final class Point {
                public final long x;
                public final long y;

                @JsonCreator
                public Point(@JsonProperty("x") long x, @JsonProperty("y") long y){
                    this.x = x;
                    this.y = y;
                }
            }
        }
    }

    public static final class Route {
        public final String name;
        public final String guide;
        public final long distance;
        public final long time;
        public final Point point[];

        @JsonCreator
        public Route(@JsonProperty("name") String name, @JsonProperty("guide") String guide, @JsonProperty("distance") long distance, @JsonProperty("time") long time, @JsonProperty("point") Point[] point){
            this.name = name;
            this.guide = guide;
            this.distance = distance;
            this.time = time;
            this.point = point;
        }

        public static final class Point {
            public final String name;
            public final String key;
            public final long x;
            public final long y;
            public final Guide guide;
            public final Road road;
            public final Panorama panorama;

            @JsonCreator
            public Point(@JsonProperty("name") String name, @JsonProperty("key") String key, @JsonProperty("x") long x, @JsonProperty("y") long y, @JsonProperty("guide") Guide guide, @JsonProperty("road") Road road, @JsonProperty("panorama") Panorama panorama){
                this.name = name;
                this.key = key;
                this.x = x;
                this.y = y;
                this.guide = guide;
                this.road = road;
                this.panorama = panorama;
            }

            public static final class Guide {
                public final long no;
                public final long value;
                public final String label;
                public final String name;

                @JsonCreator
                public Guide(@JsonProperty("no") long no, @JsonProperty("value") long value, @JsonProperty("label") String label, @JsonProperty("name") String name){
                    this.no = no;
                    this.value = value;
                    this.label = label;
                    this.name = name;
                }
            }

            public static final class Road {
                public final long no;
                public final long type;
                public final long congestion;
                public final long speed;
                public final long time;
                public final long cctv;
                public final String uid;
                public final long distance;

                @JsonCreator
                public Road(@JsonProperty("no") long no, @JsonProperty("type") long type, @JsonProperty("congestion") long congestion, @JsonProperty("speed") long speed, @JsonProperty("time") long time, @JsonProperty("cctv") long cctv, @JsonProperty("uid") String uid, @JsonProperty("distance") long distance){
                    this.no = no;
                    this.type = type;
                    this.congestion = congestion;
                    this.speed = speed;
                    this.time = time;
                    this.cctv = cctv;
                    this.uid = uid;
                    this.distance = distance;
                }
            }

            public static final class Panorama {
                public final String id;
                public final long pan;
                public final double lng;
                public final double lat;
                public final long tilt;

                @JsonCreator
                public Panorama(@JsonProperty("id") String id, @JsonProperty("pan") long pan, @JsonProperty("lng") double lng, @JsonProperty("lat") double lat, @JsonProperty("tilt") long tilt){
                    this.id = id;
                    this.pan = pan;
                    this.lng = lng;
                    this.lat = lat;
                    this.tilt = tilt;
                }
            }
        }
    }
}
